for i in {1..30}
do
    python read_organizer.py -i myreads.fa -o myreads_organized.txt -min $i
    gzip myreads_organized.txt
    new = $(ls -l myreads_organized.txt.gz | awk '{print $5}')
    grep -v "^>" myreads.fa | gzip > myreads.txt.gz
    old = $(ls -l myreads.txt.gz | awk '{print $5}')
    echo $i $old/$new >> score.txt
    
done
